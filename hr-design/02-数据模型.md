# 核心数据模型

## 第一期 · 核心基建 — 详细表设计

### 1. 组织管理

```sql
-- 组织架构表（支持集团-公司-事业部-中心-部门-团队等多级）
hr_organization (
  org_id          bigint       PK     -- 组织ID
  parent_id       bigint              -- 上级组织ID
  ancestors       varchar(500)        -- 祖级列表（如 0,1,2）
  org_name        varchar(100)        -- 组织名称
  org_code        varchar(50)  UNI    -- 组织编码
  org_type        char(1)             -- 组织类型（1集团 2公司 3事业部 4中心 5部门 6团队）
  org_level       int                 -- 组织层级
  leader_id       bigint              -- 负责人（关联 hr_employee）
  phone           varchar(20)         -- 联系电话
  address         varchar(200)        -- 办公地址
  establish_date  date                -- 成立日期
  order_num       int                 -- 排序号
  status          char(1)             -- 状态（0正常 1停用）
  sys_dept_id     bigint              -- 关联若依 sys_dept.dept_id（同步桥接）
)

-- 组织变更记录表（追溯组织架构调整历史）
hr_org_change (
  change_id       bigint       PK
  org_id          bigint              -- 组织ID
  change_type     char(1)             -- 变更类型（1新设 2合并 3拆分 4撤销 5更名）
  change_reason   varchar(500)        -- 变更原因
  effective_date  date                -- 生效日期
  old_parent_id   bigint              -- 原上级组织
  new_parent_id   bigint              -- 新上级组织
  process_id      varchar(64)         -- 关联审批流程实例ID
)
```

### 2. 岗位职级体系

```sql
-- 职族（如：管理族、技术族、职能族、营销族）
hr_position_category (
  category_id     bigint       PK
  category_name   varchar(50)         -- 职族名称
  category_code   varchar(30)  UNI    -- 职族编码
  description     varchar(500)        -- 描述
  status          char(1)
)

-- 职级（如：P1-P10、M1-M6）
hr_position_level (
  level_id        bigint       PK
  category_id     bigint       FK     -- 所属职族
  level_name      varchar(50)         -- 职级名称（如 P7 高级工程师）
  level_code      varchar(30)  UNI    -- 职级编码
  level_rank      int                 -- 等级序号（用于排序和晋升路径）
  salary_min      decimal(10,2)       -- 薪资下限（参考值）
  salary_max      decimal(10,2)       -- 薪资上限
  description     varchar(500)
  status          char(1)
)

-- HR 岗位表（比 sys_post 更丰富）
hr_position (
  position_id     bigint       PK
  position_name   varchar(100)        -- 岗位名称
  position_code   varchar(50)  UNI    -- 岗位编码
  category_id     bigint       FK     -- 所属职族
  org_id          bigint       FK     -- 所属组织
  headcount       int                 -- 编制人数
  current_count   int                 -- 在岗人数
  responsibilities text               -- 岗位职责
  requirements    text                -- 任职要求
  status          char(1)
  sys_post_id     bigint              -- 关联若依 sys_post（桥接）
)
```

### 3. 员工档案（核心表）

```sql
-- 员工主表
hr_employee (
  employee_id     bigint       PK
  employee_no     varchar(30)  UNI    -- 工号
  user_id         bigint       UNI    -- 关联 sys_user.user_id（桥接）
  name            varchar(50)         -- 姓名
  gender          char(1)             -- 性别
  id_card         varchar(18)  UNI    -- 身份证号
  birthday        date                -- 出生日期
  phone           varchar(20)         -- 手机号
  email           varchar(100)        -- 邮箱
  photo           varchar(200)        -- 照片路径
  org_id          bigint       FK     -- 所属组织
  position_id     bigint       FK     -- 岗位
  level_id        bigint       FK     -- 职级
  direct_leader_id bigint             -- 直属上级（员工ID）
  employee_type   char(1)             -- 用工类型（1正式 2实习 3劳务派遣 4外包）
  employee_status char(1)             -- 状态（0待入职 1试用 2正式 3调岗中 4离职中 5已离职）
  hire_date       date                -- 入职日期
  probation_end   date                -- 试用期结束日期
  regular_date    date                -- 转正日期
  leave_date      date                -- 离职日期
  leave_reason    varchar(500)        -- 离职原因
  work_location   varchar(200)        -- 工作地点
  nationality     varchar(20)         -- 民族
  native_place    varchar(100)        -- 籍贯
  marital_status  char(1)             -- 婚姻状况
  political       varchar(20)         -- 政治面貌
  highest_edu     varchar(20)         -- 最高学历
)

-- 教育经历
hr_employee_education (
  edu_id          bigint       PK
  employee_id     bigint       FK
  school          varchar(100)        -- 学校名称
  major           varchar(100)        -- 专业
  degree          varchar(20)         -- 学位
  education       varchar(20)         -- 学历（本科/硕士/博士）
  start_date      date
  end_date        date
)

-- 工作经历
hr_employee_work_history (
  history_id      bigint       PK
  employee_id     bigint       FK
  company         varchar(100)        -- 公司名称
  position        varchar(50)         -- 职位
  start_date      date
  end_date        date
  leave_reason    varchar(200)        -- 离职原因
  reference_name  varchar(50)         -- 证明人
  reference_phone varchar(20)         -- 证明人电话
)

-- 紧急联系人
hr_employee_contact (
  contact_id      bigint       PK
  employee_id     bigint       FK
  name            varchar(50)
  relation        varchar(20)         -- 关系
  phone           varchar(20)
  address         varchar(200)
)
```

### 4. 合同管理

```sql
hr_contract (
  contract_id     bigint       PK
  employee_id     bigint       FK
  contract_no     varchar(50)  UNI    -- 合同编号
  contract_type   char(1)             -- 类型（1固定期限 2无固定期限 3实习协议 4劳务协议）
  start_date      date                -- 合同开始日期
  end_date        date                -- 合同结束日期
  sign_date       date                -- 签订日期
  probation_months int                -- 试用期月数
  renewal_count   int          DEF 0  -- 续签次数
  salary_amount   decimal(10,2)       -- 合同薪资
  status          char(1)             -- 状态（0待签 1生效中 2已到期 3已终止）
  attachment      varchar(500)        -- 合同附件路径
  process_id      varchar(64)         -- 审批流程实例ID
  remark          varchar(500)
)
```

### 5. 员工异动表（入转调离统一记录）

```sql
hr_employee_change (
  change_id       bigint       PK
  employee_id     bigint       FK
  change_type     char(1)             -- 异动类型（1入职 2转正 3调岗 4晋升 5降级 6离职）
  -- 变更内容
  old_org_id      bigint              -- 原组织
  new_org_id      bigint              -- 新组织
  old_position_id bigint              -- 原岗位
  new_position_id bigint              -- 新岗位
  old_level_id    bigint              -- 原职级
  new_level_id    bigint              -- 新职级
  change_reason   varchar(500)        -- 异动原因
  effective_date  date                -- 生效日期
  process_id      varchar(64)         -- 关联审批流程实例ID
  status          char(1)             -- 状态（0审批中 1已生效 2已驳回 3已撤销）
  remark          varchar(500)
)
```
